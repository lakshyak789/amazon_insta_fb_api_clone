API Assignment Spec — Amazon, Facebook & Instagram Clones
Conventions (apply to all 3 projects)
Base URL: https://api.example.com/v1


Auth: Bearer JWT in Authorization: Bearer <token> unless noted public.


Content-Type: application/json (file uploads use multipart/form-data).


Pagination: ?page=<int>&limit=<int>; responses wrap lists as { "data": [...], "page": 1, "limit": 20, "total": 123 }.


Sorting & filtering: ?sort=field,-field&filter[field]=value.


Timestamps: ISO 8601 UTC created_at, updated_at.


Errors:

 { "error": { "code": "RESOURCE_NOT_FOUND", "message": "Not found", "details": [] } }


ID format: UUID v4 unless otherwise stated.


Soft-deletes: deleted_at nullable where relevant.


Field naming: snake_case in DB, camelCase in JSON.



Project A — Amazon (E-commerce) Clone
Core Entities & DB Schema (DDL-style)
-- Users & Auth
CREATE TABLE users (
  id UUID PRIMARY KEY,
  email TEXT UNIQUE NOT NULL,
  password_hash TEXT NOT NULL,
  full_name TEXT NOT NULL,
  phone TEXT,
  role TEXT NOT NULL DEFAULT 'customer', -- customer|admin|seller
  is_active BOOLEAN NOT NULL DEFAULT TRUE,
  created_at TIMESTAMPTZ NOT NULL,
  updated_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE addresses (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  label TEXT, -- "Home", "Work"
  full_name TEXT NOT NULL,
  line1 TEXT NOT NULL, line2 TEXT,
  city TEXT NOT NULL, state TEXT, postal_code TEXT NOT NULL,
  country TEXT NOT NULL,
  phone TEXT,
  is_default BOOLEAN NOT NULL DEFAULT FALSE,
  created_at TIMESTAMPTZ NOT NULL, updated_at TIMESTAMPTZ NOT NULL
);

-- Catalog
CREATE TABLE categories (
  id UUID PRIMARY KEY,
  name TEXT NOT NULL,
  slug TEXT UNIQUE NOT NULL,
  parent_id UUID REFERENCES categories(id),
  created_at TIMESTAMPTZ NOT NULL, updated_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE products (
  id UUID PRIMARY KEY,
  sku TEXT UNIQUE,
  title TEXT NOT NULL,
  slug TEXT UNIQUE NOT NULL,
  description TEXT,
  brand TEXT,
  price_cents INT NOT NULL,
  currency CHAR(3) NOT NULL DEFAULT 'USD',
  category_id UUID REFERENCES categories(id),
  avg_rating NUMERIC(2,1) DEFAULT 0,
  review_count INT NOT NULL DEFAULT 0,
  created_at TIMESTAMPTZ NOT NULL, updated_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE product_images (
  id UUID PRIMARY KEY,
  product_id UUID REFERENCES products(id),
  url TEXT NOT NULL,
  is_primary BOOLEAN NOT NULL DEFAULT FALSE,
  created_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE inventory (
  id UUID PRIMARY KEY,
  product_id UUID REFERENCES products(id) UNIQUE,
  quantity INT NOT NULL DEFAULT 0,
  reserved INT NOT NULL DEFAULT 0,
  warehouse TEXT,
  updated_at TIMESTAMPTZ NOT NULL
);

-- Cart & Wishlist
CREATE TABLE carts (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id) UNIQUE,
  created_at TIMESTAMPTZ NOT NULL, updated_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE cart_items (
  id UUID PRIMARY KEY,
  cart_id UUID REFERENCES carts(id),
  product_id UUID REFERENCES products(id),
  quantity INT NOT NULL CHECK (quantity > 0),
  price_cents INT NOT NULL, currency CHAR(3) NOT NULL DEFAULT 'USD',
  created_at TIMESTAMPTZ NOT NULL, updated_at TIMESTAMPTZ NOT NULL,
  UNIQUE(cart_id, product_id)
);

CREATE TABLE wishlists (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id) UNIQUE,
  created_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE wishlist_items (
  id UUID PRIMARY KEY,
  wishlist_id UUID REFERENCES wishlists(id),
  product_id UUID REFERENCES products(id),
  created_at TIMESTAMPTZ NOT NULL,
  UNIQUE(wishlist_id, product_id)
);

-- Orders & Payments
CREATE TABLE orders (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  address_id UUID REFERENCES addresses(id),
  status TEXT NOT NULL, -- created|paid|shipped|delivered|cancelled|refunded
  subtotal_cents INT NOT NULL,
  shipping_cents INT NOT NULL,
  tax_cents INT NOT NULL,
  total_cents INT NOT NULL,
  currency CHAR(3) NOT NULL DEFAULT 'USD',
  payment_id UUID,
  placed_at TIMESTAMPTZ,
  created_at TIMESTAMPTZ NOT NULL, updated_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE order_items (
  id UUID PRIMARY KEY,
  order_id UUID REFERENCES orders(id),
  product_id UUID REFERENCES products(id),
  title TEXT NOT NULL,
  price_cents INT NOT NULL,
  quantity INT NOT NULL,
  currency CHAR(3) NOT NULL DEFAULT 'USD'
);

CREATE TABLE payments (
  id UUID PRIMARY KEY,
  order_id UUID REFERENCES orders(id) UNIQUE,
  provider TEXT NOT NULL, -- stripe|razorpay|paypal
  provider_ref TEXT,
  status TEXT NOT NULL, -- initiated|authorized|captured|failed|refunded
  amount_cents INT NOT NULL, currency CHAR(3) NOT NULL DEFAULT 'USD',
  created_at TIMESTAMPTZ NOT NULL, updated_at TIMESTAMPTZ NOT NULL
);

-- Reviews
CREATE TABLE reviews (
  id UUID PRIMARY KEY,
  product_id UUID REFERENCES products(id),
  user_id UUID REFERENCES users(id),
  rating INT CHECK (rating BETWEEN 1 AND 5),
  title TEXT, body TEXT,
  helpful_count INT NOT NULL DEFAULT 0,
  created_at TIMESTAMPTZ NOT NULL, updated_at TIMESTAMPTZ NOT NULL,
  UNIQUE(product_id, user_id)
);

-- Admin Ops & Logs
CREATE TABLE audit_logs (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  action TEXT NOT NULL,
  target_type TEXT, target_id UUID,
  meta JSONB,
  created_at TIMESTAMPTZ NOT NULL
);

Project B — Facebook (Social Network) Clone
Core Entities & DB Schema
CREATE TABLE users (
  id UUID PRIMARY KEY, email TEXT UNIQUE NOT NULL, password_hash TEXT NOT NULL,
  full_name TEXT NOT NULL, username TEXT UNIQUE NOT NULL,
  bio TEXT, avatar_url TEXT, cover_url TEXT,
  is_active BOOLEAN NOT NULL DEFAULT TRUE,
  created_at TIMESTAMPTZ NOT NULL, updated_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE friendships (
  id UUID PRIMARY KEY,
  requester_id UUID REFERENCES users(id),
  addressee_id UUID REFERENCES users(id),
  status TEXT NOT NULL, -- pending|accepted|declined|blocked
  created_at TIMESTAMPTZ NOT NULL, updated_at TIMESTAMPTZ NOT NULL,
  UNIQUE(requester_id, addressee_id)
);

CREATE TABLE posts (
  id UUID PRIMARY KEY,
  author_id UUID REFERENCES users(id),
  body TEXT, privacy TEXT NOT NULL DEFAULT 'friends', -- public|friends|private
  media_url TEXT, media_type TEXT, -- image|video
  like_count INT NOT NULL DEFAULT 0, comment_count INT NOT NULL DEFAULT 0,
  created_at TIMESTAMPTZ NOT NULL, updated_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE comments (
  id UUID PRIMARY KEY,
  post_id UUID REFERENCES posts(id),
  author_id UUID REFERENCES users(id),
  body TEXT NOT NULL,
  parent_id UUID REFERENCES comments(id),
  created_at TIMESTAMPTZ NOT NULL, updated_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE reactions (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  target_type TEXT NOT NULL, -- post|comment
  target_id UUID NOT NULL,
  type TEXT NOT NULL, -- like|love|haha|wow|sad|angry
  created_at TIMESTAMPTZ NOT NULL,
  UNIQUE(user_id, target_type, target_id)
);

CREATE TABLE groups (
  id UUID PRIMARY KEY,
  name TEXT NOT NULL, slug TEXT UNIQUE NOT NULL,
  description TEXT, privacy TEXT NOT NULL DEFAULT 'public', -- public|private
  owner_id UUID REFERENCES users(id),
  created_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE group_members (
  id UUID PRIMARY KEY,
  group_id UUID REFERENCES groups(id),
  user_id UUID REFERENCES users(id),
  role TEXT NOT NULL DEFAULT 'member', -- member|admin|moderator
  status TEXT NOT NULL DEFAULT 'joined', -- joined|pending
  created_at TIMESTAMPTZ NOT NULL, UNIQUE(group_id,user_id)
);

CREATE TABLE pages (
  id UUID PRIMARY KEY,
  name TEXT NOT NULL, slug TEXT UNIQUE NOT NULL,
  category TEXT, about TEXT,
  owner_id UUID REFERENCES users(id),
  follower_count INT NOT NULL DEFAULT 0,
  created_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE messages (
  id UUID PRIMARY KEY,
  thread_id UUID NOT NULL,
  sender_id UUID REFERENCES users(id),
  body TEXT, media_url TEXT,
  created_at TIMESTAMPTZ NOT NULL
);
CREATE INDEX ON messages(thread_id, created_at);

CREATE TABLE notifications (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  type TEXT NOT NULL, -- friend_request|comment|reaction|group_invite|message
  ref_type TEXT, ref_id UUID,
  is_read BOOLEAN NOT NULL DEFAULT FALSE,
  created_at TIMESTAMPTZ NOT NULL
);

Project C — Instagram (Photo/Video Sharing) Clone
Core Entities & DB Schema
CREATE TABLE users (
  id UUID PRIMARY KEY, email TEXT UNIQUE NOT NULL, password_hash TEXT NOT NULL,
  username TEXT UNIQUE NOT NULL, full_name TEXT NOT NULL,
  bio TEXT, avatar_url TEXT,
  is_verified BOOLEAN NOT NULL DEFAULT FALSE,
  created_at TIMESTAMPTZ NOT NULL, updated_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE follows (
  id UUID PRIMARY KEY,
  follower_id UUID REFERENCES users(id),
  followee_id UUID REFERENCES users(id),
  status TEXT NOT NULL DEFAULT 'accepted', -- accepted|requested|blocked
  created_at TIMESTAMPTZ NOT NULL,

CREATE TABLE media (
  id UUID PRIMARY KEY,
  owner_id UUID REFERENCES users(id),
  type TEXT NOT NULL, -- image|video|carousel
  urls JSONB NOT NULL, -- {items:[{url,width,height}], thumb:"..."}
  caption TEXT, location TEXT,
  like_count INT NOT NULL DEFAULT 0, comment_count INT NOT NULL DEFAULT 0,
  is_archived BOOLEAN NOT NULL DEFAULT FALSE,
  created_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE story (
  id UUID PRIMARY KEY,
  owner_id UUID REFERENCES users(id),
  media_url TEXT NOT NULL, expires_at TIMESTAMPTZ NOT NULL,
  view_count INT NOT NULL DEFAULT 0, created_at TIMESTAMPTZ NOT NULL

CREATE TABLE comments (
  id UUID PRIMARY KEY,
  media_id UUID REFERENCES media(id),
  user_id UUID REFERENCES users(id),
  body TEXT NOT NULL,
  parent_id UUID REFERENCES comments(id),
  created_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE likes (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),

  UNIQUE(user_id, media_id)

CREATE TABLE tags (
  id UUID PRIMARY KEY,

CREATE TABLE media_tags (
  id UUID PRIMARY KEY,
  media_id UUID REFERENCES media(id),
  tag_id UUID REFERENCES tags(id),

);

CREATE TABLE direct_threads (
  id UUID PRIMARY KEY,
  is_group BOOLEAN NOT NULL DEFAULT FALSE,
  created_at TIMESTAMPTZ NOT NULL
);
CREATE TABLE direct_participants (
  id UUID PRIMARY KEY,
  thread_id UUID REFERENCES direct_threads(id),
  user_id UUID REFERENCES users(id),
  role TEXT NOT NULL DEFAULT 'member',
  UNIQUE(thread_id,user_id)
);
CREATE TABLE direct_messages (
  id UUID PRIMARY KEY,
  thread_id UUID REFERENCES direct_threads(id),
  sender_id UUID REFERENCES users(id),
  body TEXT, media_url TEXT,
  created_at TIMESTAMPTZ NOT NULL
);

CREATE TABLE notifications (
  id UUID PRIMARY KEY,
  user_id UUID REFERENCES users(id),
  type TEXT NOT NULL, -- like|comment|follow|mention|dm
  ref_type TEXT, ref_id UUID, is_read BOOLEAN NOT NULL DEFAULT FALSE,
  created_at TIMESTAMPTZ NOT NULL
);

Security, Validation & Notes (for the developer)
Auth flows: Registration returns access + refresh tokens. Use short-lived access tokens and refresh rotation.


Rate limiting: Suggest X-RateLimit-* headers; 429 body uses standard error shape.


Idempotency: Payments and order mutations accept Idempotency-Key header.


Input validation:


Emails RFC 5322; password min 8 chars, 1 upper, 1 lower, 1 digit.


Price fields are integer cents.


Files: max size, mime allowlists.


Privacy checks:


Facebook/Instagram visibility (friends/followers) must be enforced at query time.


Search:


Provide q full-text across names, titles, captions; consider trigram/FTS indexes.
